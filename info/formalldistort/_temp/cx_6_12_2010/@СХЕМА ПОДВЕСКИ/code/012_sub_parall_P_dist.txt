sub parall_P_dist(X,Y,Z,A,B,C,znak,XD,YD,ZD,DX,DY,DZ)
dim S#,T#
S=1/POLE(ko_n) 
do while ((XD-X)^2+(YD-Y)^2+(ZD-Z)^2)^0.5 < 1
T= S*znak/B
XD = X + T*A
YD = Y + T*B
ZD = Z + T*C
S=S+1/POLE(ko_n)
loop
DX=XD-X
DY=YD-Y
DZ=ZD-Z

end sub

X,Y,Z - координаты точки центра масс колеса
A,B,C - составляющие направляющего вектора плоскостей
XD,YD,ZD - искомые координаты точки параллельной плоскости на дистанции
DX,DY,DZ - приращение кратно которому задаются остальные точки плоскостей, путем умножения
znak - направление изменения Y при поиске точки новой плоскости (1||-1)
